# python3-module-PPyTooL
i invented this python3 module name PPyTooL where you can do loading and rainbow

# example of use in file
from PPyTooL import rainbow, loading

rainbow("Test of rainbow test and loading screen")
loading("Loading...", duration=5, interval=0.2)

0.1 = fast
0.2 = normal
bigger the number slower it be
